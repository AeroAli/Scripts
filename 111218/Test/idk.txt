def enumProcs(host):
	command = ["rpcclient", host, "-U", username + "%" + password, "-c"]

	procsDict = splitIntoDict(subprocess.run(command+["enumprocs"], encoding="ascii", stdout=subprocess.PIPE).stdout.splitlines())
    
    return procsDict
#end enumProcs
